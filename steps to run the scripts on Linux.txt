steps to run the scripts on Linux 

sudo apt update

sudo apt install -y python3 python3-pip curl sudo


pip3 install --user "fastapi[standard]"

# Find the primary interface name (usually "ens33" or "eth0")
ip a

# setup visudo to ignore password prompts when running the script
#run which tc and should return yourusername ALL=(ALL) NOPASSWD: /usr/sbin/tc
#also add this for stress ng
yourusername ALL=(ALL) NOPASSWD: /usr/bin/pkill

#add this to visudo at the end between these lines 
sudo visudo
yourusername ALL=(ALL) NOPASSWD: /usr/sbin/tc

--------------------------------------
# User privilege specification
root    ALL=(ALL:ALL) ALL

# Members of the admin group may gain root privileges

# Allow members of group sudo to execute any command

test ALL=(ALL) NOPASSWD: /usr/sbin/tc
